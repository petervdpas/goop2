#!/usr/bin/env bash
BASE="http://localhost:8787"

# Background heartbeat
(while true; do
  curl -sf -X POST "$BASE/api/cluster/heartbeat" \
    -H 'Content-Type: application/json' \
    -d '{"stats":{"executor":"bash","pid":'$$'}}' >/dev/null 2>&1
  sleep 10
done) &

while true; do
  # 1. Poll for pending jobs
  JOBS=$(curl -sf "$BASE/api/cluster/job")
  JOB_ID=$(echo "$JOBS" | python3 -c "
    import sys,json; p=json.load(sys.stdin).get('pending',[])
    print(p[0]['job']['id'] if p else '')")
  [ -z "$JOB_ID" ] && sleep 2 && continue

  # 2. Accept the job
  curl -sf -X POST "$BASE/api/cluster/accept" \
    -H 'Content-Type: application/json' \
    -d "{\"job_id\":\"$JOB_ID\"}"

  # 3. Do work + report progress
  for pct in 25 50 75 100; do
    sleep 1  # your work here
    curl -sf -X POST "$BASE/api/cluster/progress" \
      -H 'Content-Type: application/json' \
      -d "{\"job_id\":\"$JOB_ID\",\"percent\":$pct,\"message\":\"$pct%\"}"
  done

  # 4. Report result
  curl -sf -X POST "$BASE/api/cluster/result" \
    -H 'Content-Type: application/json' \
    -d "{\"job_id\":\"$JOB_ID\",\"success\":true,\"result\":{\"done\":true}}"
done
import time, threading, os, requests

BASE = "http://localhost:8787"

# Background heartbeat
def heartbeat():
    while True:
        try: requests.post(f"{BASE}/api/cluster/heartbeat",
                 json={"stats": {"executor": "python", "pid": os.getpid()}})
        except: pass
        time.sleep(10)

threading.Thread(target=heartbeat, daemon=True).start()

while True:
    # 1. Poll for pending jobs
    jobs = requests.get(f"{BASE}/api/cluster/job").json()
    pending = jobs.get("pending", [])
    if not pending:
        time.sleep(2); continue

    job = pending[0]["job"]
    job_id = job["id"]

    # 2. Accept the job
    requests.post(f"{BASE}/api/cluster/accept", json={"job_id": job_id})

    # 3. Do work + report progress
    for pct in range(25, 101, 25):
        time.sleep(1)  # your work here
        requests.post(f"{BASE}/api/cluster/progress",
            json={"job_id": job_id, "percent": pct, "message": f"{pct}%"})

    # 4. Report result
    requests.post(f"{BASE}/api/cluster/result",
        json={"job_id": job_id, "success": True, "result": {"done": True}})
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "net/http"
    "os"
    "time"
)

var base = "http://localhost:8787"

func postJSON(url string, body any) {
    b, _ := json.Marshal(body)
    http.Post(url, "application/json", bytes.NewReader(b))
}

func main() {
    // Background heartbeat
    go func() {
        for {
            postJSON(base+"/api/cluster/heartbeat", map[string]any{
                "stats": map[string]any{"executor": "go", "pid": os.Getpid()},
            })
            time.Sleep(10 * time.Second)
        }
    }()

    for {
        // 1. Poll for pending jobs
        resp, err := http.Get(base + "/api/cluster/job")
        if err != nil { time.Sleep(2 * time.Second); continue }
        var data struct {
            Pending []struct{ Job struct{ ID, Type string } }
        }
        json.NewDecoder(resp.Body).Decode(&data)
        resp.Body.Close()
        if len(data.Pending) == 0 { time.Sleep(2 * time.Second); continue }

        jobID := data.Pending[0].Job.ID

        // 2. Accept the job
        postJSON(base+"/api/cluster/accept", map[string]any{"job_id": jobID})

        // 3. Do work + report progress
        for pct := 25; pct <= 100; pct += 25 {
            time.Sleep(1 * time.Second) // your work here
            postJSON(base+"/api/cluster/progress", map[string]any{
                "job_id": jobID, "percent": pct,
                "message": fmt.Sprintf("%d%%", pct),
            })
        }

        // 4. Report result
        postJSON(base+"/api/cluster/result", map[string]any{
            "job_id": jobID, "success": true,
            "result": map[string]any{"done": true},
        })
    }
}

static const char *BASE = "http://localhost:8787";

int post_json(const char *url, const char *body) {
    CURL *c = curl_easy_init();
    struct curl_slist *h = curl_slist_append(NULL, "Content-Type: application/json");
    curl_easy_setopt(c, CURLOPT_URL, url);
    curl_easy_setopt(c, CURLOPT_POSTFIELDS, body);
    curl_easy_setopt(c, CURLOPT_HTTPHEADER, h);
    curl_easy_perform(c);
    curl_slist_free_all(h);
    curl_easy_cleanup(c);
    return 0;
}

void *heartbeat_thread(void *arg) {
    char url[256], body[256];
    snprintf(url, sizeof(url), "%s/api/cluster/heartbeat", BASE);
    snprintf(body, sizeof(body),
        "{\"stats\":{\"executor\":\"c\",\"pid\":%d}}", (int)getpid());
    for (;;) { post_json(url, body); sleep(10); }
    return NULL;
}

}
