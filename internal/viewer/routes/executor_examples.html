#!/usr/bin/env bash
#
# cluster-worker.sh — goop2 cluster binary (daemon mode)
#
# Protocol: reads JSON jobs from stdin (one per line),
#           writes JSON responses to stdout (one per line).
#
# Set as binary path on worker peer:
#   POST /api/cluster/binary {"path":"/path/to/cluster-worker.sh","mode":"daemon"}
#
# Test manually:
#   echo '{"job_id":"__check__","type":"__check__","payload":{}}' | ./cluster-worker.sh
#   echo '{"job_id":"test-1","type":"echo","payload":{"message":"hello"}}' | ./cluster-worker.sh

set -euo pipefail

log() { echo "[$(date +%H:%M:%S)] $*" >&2; }

handle_check() {
  echo '{"status":"done","result":{"types":["echo","count","sleep"],"version":"1.0","capacity":1}}'
}

handle_echo() {
  local job_id="$1" message="$2"
  echo "{\"status\":\"progress\",\"pct\":50,\"msg\":\"echoing...\"}"
  echo "{\"status\":\"done\",\"result\":{\"echoed\":\"$message\"}}"
}

handle_count() {
  local job_id="$1" n="$2"
  if [ "$n" -le 0 ] 2>/dev/null; then
    echo "{\"status\":\"error\",\"error\":\"count must be a positive integer, got: $n\"}"
    return
  fi
  for ((i=1; i<=n; i++)); do
    pct=$(( i * 100 / n ))
    echo "{\"status\":\"progress\",\"pct\":$pct,\"msg\":\"counted $i of $n\"}"
    sleep 0.2
  done
  echo "{\"status\":\"done\",\"result\":{\"counted\":$n}}"
}

handle_sleep() {
  local job_id="$1" seconds="$2"
  local steps=5
  for ((i=1; i<=steps; i++)); do
    sleep "$(echo "$seconds / $steps" | bc -l)"
    pct=$(( i * 100 / steps ))
    echo "{\"status\":\"progress\",\"pct\":$pct,\"msg\":\"sleeping $i/$steps\"}"
  done
  echo "{\"status\":\"done\",\"result\":{\"slept_seconds\":$seconds}}"
}

log "cluster-worker.sh started, reading jobs from stdin"

while IFS= read -r line; do
  JOB_ID=$(echo "$line" | python3 -c "import sys,json; print(json.load(sys.stdin).get('job_id',''))")
  TYPE=$(echo "$line" | python3 -c "import sys,json; print(json.load(sys.stdin).get('type',''))")
  log "received job $JOB_ID type=$TYPE"

  case "$TYPE" in
    __check__)
      handle_check
      ;;
    echo)
      MSG=$(echo "$line" | python3 -c "import sys,json; print(json.load(sys.stdin).get('payload',{}).get('message',''))")
      handle_echo "$JOB_ID" "$MSG"
      ;;
    count)
      N=$(echo "$line" | python3 -c "import sys,json; print(json.load(sys.stdin).get('payload',{}).get('n',0))")
      handle_count "$JOB_ID" "$N"
      ;;
    sleep)
      SECS=$(echo "$line" | python3 -c "import sys,json; print(json.load(sys.stdin).get('payload',{}).get('seconds',1))")
      handle_sleep "$JOB_ID" "$SECS"
      ;;
    *)
      echo "{\"status\":\"error\",\"error\":\"unknown job type: $TYPE\"}"
      ;;
  esac
done

log "stdin closed, exiting"

#!/usr/bin/env python3
"""
cluster_worker.py — goop2 cluster binary (daemon mode)

Protocol: reads JSON jobs from stdin (one per line),
          writes JSON responses to stdout (one per line).

Set as binary path on worker peer:
  POST /api/cluster/binary {"path":"/path/to/cluster_worker.py","mode":"daemon"}

Test manually:
  echo '{"job_id":"__check__","type":"__check__","payload":{}}' | python3 cluster_worker.py
  echo '{"job_id":"t1","type":"wordcount","payload":{"text":"hello world foo"}}' | python3 cluster_worker.py
"""
import sys
import json
import time
import hashlib

def emit(obj):
    print(json.dumps(obj), flush=True)

def progress(pct, msg):
    emit({"status": "progress", "pct": pct, "msg": msg})

def done(result):
    emit({"status": "done", "result": result})

def fail(error):
    emit({"status": "error", "error": error})

def handle_check():
    done({
        "types": ["echo", "wordcount", "hash", "sleep"],
        "version": "1.0",
        "capacity": 2,
    })

def handle_echo(payload):
    message = payload.get("message", "")
    progress(50, "echoing...")
    done({"echoed": message, "length": len(message)})

def handle_wordcount(payload):
    text = payload.get("text", "")
    if not text:
        fail("payload.text is required")
        return

    words = text.split()
    total = len(words)
    counts = {}
    for i, word in enumerate(words):
        w = word.lower().strip(".,!?;:")
        counts[w] = counts.get(w, 0) + 1
        pct = (i + 1) * 100 // total
        if pct % 20 == 0:
            progress(pct, f"counted {i+1}/{total} words")

    top = sorted(counts.items(), key=lambda x: -x[1])[:10]
    done({
        "total_words": total,
        "unique_words": len(counts),
        "top_10": [{"word": w, "count": c} for w, c in top],
    })

def handle_hash(payload):
    text = payload.get("text", "")
    algorithm = payload.get("algorithm", "sha256")

    progress(25, f"hashing with {algorithm}")

    try:
        h = hashlib.new(algorithm)
    except ValueError:
        fail(f"unsupported algorithm: {algorithm}")
        return

    h.update(text.encode())
    progress(75, "finalizing")
    done({"algorithm": algorithm, "hash": h.hexdigest(), "input_length": len(text)})

def handle_sleep(payload):
    seconds = payload.get("seconds", 1)
    if not isinstance(seconds, (int, float)) or seconds <= 0:
        fail(f"seconds must be positive, got: {seconds}")
        return

    steps = 10
    for i in range(1, steps + 1):
        time.sleep(seconds / steps)
        progress(i * 100 // steps, f"sleeping {i}/{steps}")
    done({"slept_seconds": seconds})

HANDLERS = {
    "__check__": lambda _: handle_check(),
    "echo": handle_echo,
    "wordcount": handle_wordcount,
    "hash": handle_hash,
    "sleep": handle_sleep,
}

def main():
    sys.stderr.write("cluster_worker.py started, reading jobs from stdin\n")
    for line in sys.stdin:
        line = line.strip()
        if not line:
            continue

        try:
            job = json.loads(line)
        except json.JSONDecodeError as e:
            fail(f"invalid JSON: {e}")
            continue

        job_id = job.get("job_id", "?")
        job_type = job.get("type", "")
        payload = job.get("payload", {})
        sys.stderr.write(f"received job {job_id} type={job_type}\n")

        handler = HANDLERS.get(job_type)
        if handler is None:
            fail(f"unknown job type: {job_type}")
            continue

        try:
            handler(payload)
        except Exception as e:
            fail(f"unhandled error: {e}")

    sys.stderr.write("stdin closed, exiting\n")

if __name__ == "__main__":
    main()

package main

import (
	"bufio"
	"crypto/sha256"
	"encoding/hex"
	"encoding/json"
	"fmt"
	"os"
	"sort"
	"strings"
	"time"
)

// cluster_worker.go — goop2 cluster binary (daemon mode)
//
// Protocol: reads JSON jobs from stdin (one per line),
//           writes JSON responses to stdout (one per line).
//
// Build:
//   go build -o cluster-worker cluster_worker.go
//
// Set as binary path on worker peer:
//   POST /api/cluster/binary {"path":"/path/to/cluster-worker","mode":"daemon"}
//
// Test manually:
//   echo '{"job_id":"__check__","type":"__check__","payload":{}}' | ./cluster-worker
//   echo '{"job_id":"t1","type":"wordcount","payload":{"text":"hello world foo"}}' | ./cluster-worker

type JobInput struct {
	JobID   string         `json:"job_id"`
	Type    string         `json:"type"`
	Payload map[string]any `json:"payload"`
}

func emit(v any) {
	json.NewEncoder(os.Stdout).Encode(v)
}

func progress(pct int, msg string) {
	emit(map[string]any{"status": "progress", "pct": pct, "msg": msg})
}

func done(result map[string]any) {
	emit(map[string]any{"status": "done", "result": result})
}

func fail(err string) {
	emit(map[string]any{"status": "error", "error": err})
}

func handleCheck() {
	done(map[string]any{
		"types":    []string{"echo", "wordcount", "hash", "sleep"},
		"version":  "1.0",
		"capacity": 4,
	})
}

func handleEcho(payload map[string]any) {
	msg, _ := payload["message"].(string)
	progress(50, "echoing...")
	done(map[string]any{"echoed": msg, "length": len(msg)})
}

func handleWordcount(payload map[string]any) {
	text, _ := payload["text"].(string)
	if text == "" {
		fail("payload.text is required")
		return
	}

	words := strings.Fields(text)
	total := len(words)
	counts := map[string]int{}

	for i, word := range words {
		w := strings.ToLower(strings.Trim(word, ".,!?;:"))
		counts[w]++
		pct := (i + 1) * 100 / total
		if pct%20 == 0 {
			progress(pct, fmt.Sprintf("counted %d/%d words", i+1, total))
		}
	}

	type wc struct {
		Word  string `json:"word"`
		Count int    `json:"count"`
	}
	top := make([]wc, 0, len(counts))
	for w, c := range counts {
		top = append(top, wc{w, c})
	}
	sort.Slice(top, func(i, j int) bool { return top[i].Count > top[j].Count })
	if len(top) > 10 {
		top = top[:10]
	}

	done(map[string]any{
		"total_words":  total,
		"unique_words": len(counts),
		"top_10":       top,
	})
}

func handleHash(payload map[string]any) {
	text, _ := payload["text"].(string)
	progress(25, "hashing with sha256")
	h := sha256.Sum256([]byte(text))
	progress(75, "finalizing")
	done(map[string]any{
		"algorithm":    "sha256",
		"hash":         hex.EncodeToString(h[:]),
		"input_length": len(text),
	})
}

func handleSleep(payload map[string]any) {
	seconds, _ := payload["seconds"].(float64)
	if seconds <= 0 {
		fail(fmt.Sprintf("seconds must be positive, got: %v", payload["seconds"]))
		return
	}

	steps := 10
	d := time.Duration(float64(time.Second) * seconds / float64(steps))
	for i := 1; i <= steps; i++ {
		time.Sleep(d)
		progress(i*100/steps, fmt.Sprintf("sleeping %d/%d", i, steps))
	}
	done(map[string]any{"slept_seconds": seconds})
}

func main() {
	fmt.Fprintln(os.Stderr, "cluster-worker started, reading jobs from stdin")
	scanner := bufio.NewScanner(os.Stdin)
	scanner.Buffer(make([]byte, 0, 1024*1024), 1024*1024)

	for scanner.Scan() {
		var job JobInput
		if err := json.Unmarshal(scanner.Bytes(), &job); err != nil {
			fail(fmt.Sprintf("invalid JSON: %v", err))
			continue
		}

		fmt.Fprintf(os.Stderr, "received job %s type=%s\n", job.JobID, job.Type)

		switch job.Type {
		case "__check__":
			handleCheck()
		case "echo":
			handleEcho(job.Payload)
		case "wordcount":
			handleWordcount(job.Payload)
		case "hash":
			handleHash(job.Payload)
		case "sleep":
			handleSleep(job.Payload)
		default:
			fail(fmt.Sprintf("unknown job type: %s", job.Type))
		}
	}

	fmt.Fprintln(os.Stderr, "stdin closed, exiting")
}
gcc -o cluster-worker cluster_worker.c -lcjson && set as binary path
/*
 * cluster_worker.c — goop2 cluster binary (daemon mode)
 *
 * Protocol: reads JSON jobs from stdin (one per line),
 *           writes JSON responses to stdout (one per line).
 *
 * Build:
 *   gcc -o cluster-worker cluster_worker.c -lcjson
 *
 * Depends on: libcjson-dev (apt install libcjson-dev)
 *
 * Set as binary path on worker peer:
 *   POST /api/cluster/binary {"path":"/path/to/cluster-worker","mode":"daemon"}
 *
 * Test manually:
 *   echo '{"job_id":"__check__","type":"__check__","payload":{}}' | ./cluster-worker
 *   echo '{"job_id":"t1","type":"echo","payload":{"message":"hello"}}' | ./cluster-worker
 */
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <ctype.h>
#include <cjson/cJSON.h>

static void emit_raw(const char *json_str) {
    fputs(json_str, stdout);
    fputc('\n', stdout);
    fflush(stdout);
}

static void emit(cJSON *obj) {
    char *s = cJSON_PrintUnformatted(obj);
    emit_raw(s);
    free(s);
    cJSON_Delete(obj);
}

static void emit_progress(int pct, const char *msg) {
    cJSON *o = cJSON_CreateObject();
    cJSON_AddStringToObject(o, "status", "progress");
    cJSON_AddNumberToObject(o, "pct", pct);
    cJSON_AddStringToObject(o, "msg", msg);
    emit(o);
}

static void emit_done(cJSON *result) {
    cJSON *o = cJSON_CreateObject();
    cJSON_AddStringToObject(o, "status", "done");
    cJSON_AddItemToObject(o, "result", result);
    emit(o);
}

static void emit_error(const char *error) {
    cJSON *o = cJSON_CreateObject();
    cJSON_AddStringToObject(o, "status", "error");
    cJSON_AddStringToObject(o, "error", error);
    emit(o);
}

static void handle_check(void) {
    cJSON *result = cJSON_CreateObject();
    cJSON *types = cJSON_CreateArray();
    cJSON_AddItemToArray(types, cJSON_CreateString("echo"));
    cJSON_AddItemToArray(types, cJSON_CreateString("count"));
    cJSON_AddItemToArray(types, cJSON_CreateString("reverse"));
    cJSON_AddItemToObject(result, "types", types);
    cJSON_AddStringToObject(result, "version", "1.0");
    cJSON_AddNumberToObject(result, "capacity", 1);
    emit_done(result);
}

static void handle_echo(cJSON *payload) {
    cJSON *msg_item = cJSON_GetObjectItem(payload, "message");
    const char *message = msg_item ? msg_item->valuestring : "";

    emit_progress(50, "echoing...");

    cJSON *result = cJSON_CreateObject();
    cJSON_AddStringToObject(result, "echoed", message);
    cJSON_AddNumberToObject(result, "length", (double)strlen(message));
    emit_done(result);
}

static void handle_count(cJSON *payload) {
    cJSON *n_item = cJSON_GetObjectItem(payload, "n");
    if (!n_item || !cJSON_IsNumber(n_item) || n_item->valueint <= 0) {
        emit_error("payload.n must be a positive integer");
        return;
    }
    int n = n_item->valueint;

    for (int i = 1; i <= n; i++) {
        int pct = i * 100 / n;
        char msg[128];
        snprintf(msg, sizeof(msg), "counted %d of %d", i, n);
        emit_progress(pct, msg);
        usleep(100000); /* 100ms per step */
    }

    cJSON *result = cJSON_CreateObject();
    cJSON_AddNumberToObject(result, "counted", n);
    emit_done(result);
}

static void handle_reverse(cJSON *payload) {
    cJSON *text_item = cJSON_GetObjectItem(payload, "text");
    if (!text_item || !cJSON_IsString(text_item)) {
        emit_error("payload.text is required");
        return;
    }
    const char *text = text_item->valuestring;
    size_t len = strlen(text);

    emit_progress(25, "reversing...");

    char *reversed = malloc(len + 1);
    for (size_t i = 0; i < len; i++) {
        reversed[i] = text[len - 1 - i];
    }
    reversed[len] = '\0';

    emit_progress(75, "done reversing");

    cJSON *result = cJSON_CreateObject();
    cJSON_AddStringToObject(result, "reversed", reversed);
    cJSON_AddNumberToObject(result, "length", (double)len);
    free(reversed);
    emit_done(result);
}

int main(void) {
    fprintf(stderr, "cluster-worker started, reading jobs from stdin\n");

    char line[1024 * 64];
    while (fgets(line, sizeof(line), stdin)) {
        /* Strip trailing newline */
        size_t len = strlen(line);
        if (len > 0 && line[len-1] == '\n') line[len-1] = '\0';
        if (line[0] == '\0') continue;

        cJSON *job = cJSON_Parse(line);
        if (!job) {
            emit_error("invalid JSON");
            continue;
        }

        cJSON *type_item = cJSON_GetObjectItem(job, "type");
        cJSON *id_item = cJSON_GetObjectItem(job, "job_id");
        const char *type = type_item ? type_item->valuestring : "";
        const char *job_id = id_item ? id_item->valuestring : "?";
        cJSON *payload = cJSON_GetObjectItem(job, "payload");

        fprintf(stderr, "received job %s type=%s\n", job_id, type);

        if (strcmp(type, "__check__") == 0) {
            handle_check();
        } else if (strcmp(type, "echo") == 0) {
            handle_echo(payload);
        } else if (strcmp(type, "count") == 0) {
            handle_count(payload);
        } else if (strcmp(type, "reverse") == 0) {
            handle_reverse(payload);
        } else {
            char err[256];
            snprintf(err, sizeof(err), "unknown job type: %s", type);
            emit_error(err);
        }

        cJSON_Delete(job);
    }

    fprintf(stderr, "stdin closed, exiting\n");
    return 0;
}
g++ -std=c++17 -o cluster-worker cluster_worker.cpp && set as binary path
/*
 * cluster_worker.cpp — goop2 cluster binary (daemon mode)
 *
 * Protocol: reads JSON jobs from stdin (one per line),
 *           writes JSON responses to stdout (one per line).
 *
 * Build:
 *   g++ -std=c++17 -o cluster-worker cluster_worker.cpp
 *
 * Depends on: nlohmann/json (header-only, download json.hpp)
 *   wget https://github.com/nlohmann/json/releases/latest/download/json.hpp
 *
 * Set as binary path on worker peer:
 *   POST /api/cluster/binary {"path":"/path/to/cluster-worker","mode":"daemon"}
 *
 * Test manually:
 *   echo '{"job_id":"__check__","type":"__check__","payload":{}}' | ./cluster-worker
 *   echo '{"job_id":"t1","type":"fizzbuzz","payload":{"n":30}}' | ./cluster-worker
 */
#include <iostream>
#include <string>
#include <thread>
#include <chrono>
#include <algorithm>
#include <sstream>
#include <map>
#include "json.hpp"

using json = nlohmann::json;
using namespace std::chrono_literals;

void emit(const json &j) {
    std::cout << j.dump() << std::endl;
}

void progress(int pct, const std::string &msg) {
    emit({{"status", "progress"}, {"pct", pct}, {"msg", msg}});
}

void done(const json &result) {
    emit({{"status", "done"}, {"result", result}});
}

void fail(const std::string &error) {
    emit({{"status", "error"}, {"error", error}});
}

void handle_check() {
    done({
        {"types", {"echo", "fizzbuzz", "wordcount", "sleep"}},
        {"version", "1.0"},
        {"capacity", 4},
    });
}

void handle_echo(const json &payload) {
    std::string message = payload.value("message", "");
    progress(50, "echoing...");
    done({{"echoed", message}, {"length", message.size()}});
}

void handle_fizzbuzz(const json &payload) {
    int n = payload.value("n", 0);
    if (n <= 0) {
        fail("payload.n must be a positive integer");
        return;
    }

    json results = json::array();
    for (int i = 1; i <= n; i++) {
        std::string val;
        if (i % 15 == 0) val = "fizzbuzz";
        else if (i % 3 == 0) val = "fizz";
        else if (i % 5 == 0) val = "buzz";
        else val = std::to_string(i);
        results.push_back(val);

        int pct = i * 100 / n;
        if (pct % 20 == 0) {
            progress(pct, "fizzbuzz " + std::to_string(i) + "/" + std::to_string(n));
        }
    }

    int fizz = 0, buzz = 0, fizzbuzz = 0;
    for (auto &v : results) {
        if (v == "fizzbuzz") fizzbuzz++;
        else if (v == "fizz") fizz++;
        else if (v == "buzz") buzz++;
    }

    done({
        {"n", n},
        {"fizz_count", fizz},
        {"buzz_count", buzz},
        {"fizzbuzz_count", fizzbuzz},
        {"sequence", results},
    });
}

void handle_wordcount(const json &payload) {
    std::string text = payload.value("text", "");
    if (text.empty()) {
        fail("payload.text is required");
        return;
    }

    std::istringstream iss(text);
    std::string word;
    std::map<std::string, int> counts;
    int total = 0;

    while (iss >> word) {
        /* lowercase + strip punctuation */
        std::transform(word.begin(), word.end(), word.begin(), ::tolower);
        word.erase(std::remove_if(word.begin(), word.end(),
            [](char c) { return std::ispunct(c); }), word.end());
        if (!word.empty()) {
            counts[word]++;
            total++;
        }
    }

    progress(50, "counted " + std::to_string(total) + " words");

    /* top 10 */
    std::vector<std::pair<std::string, int>> sorted_counts(counts.begin(), counts.end());
    std::sort(sorted_counts.begin(), sorted_counts.end(),
        [](auto &a, auto &b) { return a.second > b.second; });
    if (sorted_counts.size() > 10) sorted_counts.resize(10);

    json top = json::array();
    for (auto &[w, c] : sorted_counts) {
        top.push_back({{"word", w}, {"count", c}});
    }

    done({
        {"total_words", total},
        {"unique_words", counts.size()},
        {"top_10", top},
    });
}

void handle_sleep(const json &payload) {
    double seconds = payload.value("seconds", 0.0);
    if (seconds <= 0) {
        fail("payload.seconds must be positive");
        return;
    }

    int steps = 10;
    auto step_duration = std::chrono::milliseconds(static_cast<int>(seconds * 1000 / steps));
    for (int i = 1; i <= steps; i++) {
        std::this_thread::sleep_for(step_duration);
        progress(i * 100 / steps,
            "sleeping " + std::to_string(i) + "/" + std::to_string(steps));
    }
    done({{"slept_seconds", seconds}});
}

int main() {
    std::cerr << "cluster-worker started, reading jobs from stdin" << std::endl;

    std::string line;
    while (std::getline(std::cin, line)) {
        if (line.empty()) continue;

        auto job = json::parse(line, nullptr, false);
        if (job.is_discarded()) {
            fail("invalid JSON");
            continue;
        }

        std::string job_id = job.value("job_id", "?");
        std::string type = job.value("type", "");
        json payload = job.value("payload", json::object());
        std::cerr << "received job " << job_id << " type=" << type << std::endl;

        if (type == "__check__") handle_check();
        else if (type == "echo") handle_echo(payload);
        else if (type == "fizzbuzz") handle_fizzbuzz(payload);
        else if (type == "wordcount") handle_wordcount(payload);
        else if (type == "sleep") handle_sleep(payload);
        else fail("unknown job type: " + type);
    }

    std::cerr << "stdin closed, exiting" << std::endl;
    return 0;
}
